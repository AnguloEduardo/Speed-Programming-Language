program main

dim a, b, c, d as word;

do
	let c = c+2;
loopuntil c<a endo;

if 1 < 3 then
	let a = 2;
else
	if (a<b) && (c>=d) || (b<>d) then
		let b = 3;
	else
		if (c <> c) then
			cls;
		else
			while a<c do
				let a = a+3;
				cls;
				while b<d do
					let b = b+3;
					cls;
					do
						let c = c+2;
					loopuntil c<a endo;
				whend;
			whend;
		cls;
		endif;
	endif;
endif;

end;
